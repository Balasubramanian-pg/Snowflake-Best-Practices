```sql
CREATE VIEW cloud_services_adjustment AS
SELECT 
    department,
    SUM(credits_used) AS total_credits_used,
    SUM(credits_used) * 0.10 AS cloud_services_adjustment
FROM 
    snowflake.account.usage_history
GROUP BY 
    department;
```
