```sql
CREATE VIEW client_app_costs AS
SELECT
  CLIENT_APPLICATION_NAME,
  SUM(CREDIT_USAGE) AS TOTAL_CREDITS_USED,
  SUM(CREDIT_QUOTA) AS TOTAL_CREDIT_QUOTA
FROM
  SNOWFLAKE.ACCOUNT_USAGE.CREDITS_USAGE
GROUP BY
  CLIENT_APPLICATION_NAME;
```
