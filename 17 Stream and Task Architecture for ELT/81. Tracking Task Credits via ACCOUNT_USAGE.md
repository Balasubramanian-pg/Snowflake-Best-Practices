### Implementation Example
CREATE VIEW task_credits_usage AS SELECT date_trunc('day', start_time) AS date, SUM(credits_used) AS total_credits_used FROM TABLE(INFORMATION_SCHEMA.TASK_HISTORY_BY_SESSION()) GROUP BY date ORDER BY date;
