### Code Source Execution
```sql

    CREATE VIEW contract_consumption_vs_capacity_remaining AS
    SELECT 
      c.contract_name,
      SUM(s.usage) AS total_consumption,
      c.capacity - SUM(s.usage) AS remaining_capacity
    FROM 
      snowflake_usage s
    JOIN 
      contracts c ON s.contract_id = c.contract_id
    GROUP BY 
      c.contract_name, c.capacity;
  
```
